create database faculdade; use faculdade; create table professor ( id_professor int primary key auto_increment, cpf varchar(18) unique not null, nome_professor varchar(100) not null, sexo char(1) not null, email varchar(100), telefone varchar(15) ); create table endereco_professor ( id_endereco int primary key auto_increment, logradouro varchar(100) not null, numero int not null, bairro varchar(100) not null, cidade varchar(50) not null, estado char(2) not null, id_professor int not null, constraint fk_endereco_professor foreign key (id_professor) references professor(id_professor) ); create table aluno ( id_aluno int primary key auto_increment, cpf varchar(18) unique not null, nome_social varchar(100), nome_aluno varchar(100) not null, data_nascimento date, sexo char(1) not null, email varchar(100), telefone varchar(15) ); create table endereco_aluno ( id_endereco int primary key auto_increment, logradouro varchar(100) not null, numero int not null, bairro varchar(100) not null, cidade varchar(50) not null, estado char(2) not null, id_aluno int not null, constraint fk_endereco_aluno foreign key (id_aluno) references aluno(id_aluno) ); create table curso ( id_curso int primary key auto_increment, nome_curso varchar(100) not null, modalidade varchar(30) not null, graduacao varchar(30), tempo_curso varchar(20), descricao text, id_professor int, constraint fk_curso_professor foreign key (id_professor) references professor(id_professor) ); create table disciplina ( id_disciplina int primary key auto_increment, nome_disciplina varchar(100) not null, descricao text, id_curso int not null, id_professor int not null, constraint fk_disciplina_curso foreign key (id_curso) references curso(id_curso), constraint fk_disciplina_professor foreign key (id_professor) references professor(id_professor) ); create table avaliacao ( id_avaliacao int primary key auto_increment, valor_nota decimal(10, 2) not null, data_avaliacao date not null, id_disciplina int not null, id_aluno int not null, constraint fk_avaliacao_disciplina foreign key (id_disciplina) references disciplina(id_disciplina), constraint fk_avaliacao_aluno foreign key (id_aluno) references aluno(id_aluno) ); create table turma ( id_turma int primary key auto_increment, semestre varchar(10) not null ); create table turma_aluno ( id_turma_aluno int primary key auto_increment, id_aluno int not null, id_curso int not null, id_turma int not null, constraint fk_turma_aluno_aluno foreign key (id_aluno) references aluno(id_aluno), constraint fk_turma_aluno_curso foreign key (id_curso) references curso(id_curso), constraint fk_turma_aluno_turma foreign key (id_turma) references turma(id_turma) );
